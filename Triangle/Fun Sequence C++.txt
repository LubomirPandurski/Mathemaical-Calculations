#include <iostream>
#include <algorithm>
#include <array>
#include <vector>

using namespace std;



int main()
{
	int Size;
	cin >> Size;
	int arr[200];
	
	for (int i = 0; i < Size; i++)
	{
		cin >> arr[i];
	}

	int sum = 0;

	for (int i = 0; i < Size; i++) {
		sum += arr[i];
	}
	int average = sum / Size;
	
	int n = sizeof(arr) / sizeof(arr[0]);
	sort(arr, arr + n, greater<int>());

	for (int i = 0; i < Size; i++) {
		if (Size <= 1) {
			cout << "No";
		}
		else if (arr[i] <= average) {
			if (arr[i] % 2 == 0) {
				cout<< arr[i];
			}
		}
		
	}
	
	cout << endl;
	return 0;
}
